let print_symbol _ is_non_term s print_params =
  print_fmt
    (if is_non_term then "<span class=\"nonterminal\">%s</span>" else "%s") s;
  print_params ()


let opt e print =
  print_string "<span class=\"option\">";
  par e print;
  print_string "</span>"

  print_string "<span class=\"ne_list\">";
  par e print;
  print_string "</span>"

  print_string "<span class=\"list\">";
  par e print;
  print_string "</span>"

let print_sep_list e nonempty print_sep print_x =
  par e (fun () ->
      print_x ();
      print_string (if nonempty then "<sup>+</sup>" else "<sup>*</sup>");
      print_string "<sub>"; print_sep (); print_string "</sub>")
