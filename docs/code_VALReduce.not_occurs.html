let rec not_occurs s = let not_occurs_pattern = let n = not_occurs s in function | Option x | List x | NEList x -> n x | Pair (x, y) | Preceded (x, y) | Terminated (x, y) | SepList (x, y) | SepNEList (x, y) -> n x && n y | SepPair (x, y, z) | Delimited (x, y, z) -> n x && n y && n z in function | Symbol (f, xs) -> f <> s && List.for_all (not_occurs s) xs | Pattern p -> not_occurs_pattern p | Modifier (x, _) -> not_occurs s x | Anonymous ps -> List.for_all (List.for_all (not_occurs s)) ps
