let add_terminal symbols {groups} = let rec add_terminal_actual symbols = function | Symbol (s, ps) -> let symbols = fold_left add_terminal_actual symbols ps in if Symbols.is_defined s symbols = None && String.uppercase_ascii s = s && ps = [] then Symbols.def_term s symbols else symbols | Modifier (a, _) -> add_terminal_actual symbols a | Anonymous gs -> fold_left add_terminal_group symbols gs and add_terminal_prod symbols = fold_left add_terminal_actual symbols and add_terminal_group symbols = fold_left add_terminal_prod symbols in fold_left add_terminal_group symbols groups
