let is_sep_nonempty_list rules r = let take = let rec aux pre n xs = match n, xs with | 0, xs -> List.rev pre, xs | _, [] -> List.rev pre, [] | n, x::xs -> aux (x :: pre) (n-1) xs in aux [] in let g = generalize r.params in let f base cons = match cons with | [] -> None | Symbol (s, xs) :: acts when alias rules r.name s && equal_params r.params xs && List.for_all (not_occurs s) acts -> let base_r, sep_r = take (List.length base) (List.rev acts) in if base = List.rev base_r then g (SepNEList (Anonymous [List.rev sep_r], Anonymous [base])) else None | _ -> begin match List.rev cons with | Symbol (s, xs) :: acts when alias rules r.name s && equal_params r.params xs && List.for_all (not_occurs s) acts -> let base', sep = take (List.length base) (List.rev acts) in if base = base' then g (SepNEList (Anonymous [sep], Anonymous [base])) else None | _ -> None end in match r.prods with | [x; y] -> f x y @@ f y x | _ -> None
