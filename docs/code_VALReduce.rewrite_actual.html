let rec rewrite_actual rws = function | Symbol (s, xs) -> let xs = List.map (rewrite_actual rws) xs in begin try let rw = M.find s rws in subst rw xs with Not_found -> Symbol (s, xs) end | Pattern p -> Pattern (rewrite_pattern rws p) | Modifier (x, m) -> Modifier (rewrite_actual rws x, m) | Anonymous ps -> Anonymous (List.map (rewrite_production rws) ps)
(** Inline the recognized patterns in a pattern. *)
and rewrite_pattern rws = let r = rewrite_actual rws in function | Option x -> Option (r x) | Pair (x, y) -> Pair (r x, r y) | SepPair (x, sep, y) -> SepPair (r x, r sep, r y) | Preceded (o, x) -> Preceded (r o, r x) | Terminated (x, c) -> Terminated (r x, r c) | Delimited (o, x, c) -> Delimited (r o, r x, r c) | List x -> List (r x) | NEList x -> NEList (r x) | SepList (sep, x) -> SepList (r sep, r x) | SepNEList (sep, x) -> SepNEList (r sep, r x)
(** Inline the recognized patterns in a production. *)
and rewrite_production rws = List.map (rewrite_actual rws)
